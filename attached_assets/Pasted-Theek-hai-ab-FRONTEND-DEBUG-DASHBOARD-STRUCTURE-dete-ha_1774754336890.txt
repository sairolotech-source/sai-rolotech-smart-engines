export async function runManualMode(payload) {
  const res = await fetch("/api/manual-mode", {
    method: "POST",

import { useState } from "react";

export default function UploadForm({ onRun }) {
  const [form, setForm] = useState({
    bend_count: 6,
    section_width_mm: 100,
    section_height_mm: 50,
    thickness: 0.8,
    material: "GI",
    profile_type: "lipped_channel",
  });

  function updateField(key, value) {
    setForm((prev) => ({ ...prev, [key]: value }));
  }

  function submit(e) {
    e.preventDefault();
    onRun(form);
  }

  return (
    <form onSubmit={submit} className="grid gap-3 rounded-2xl border p-4 shadow-sm bg-white">
      <div className="text-lg font-semibold">Input Panel</div>

      <input
        className="border rounded-xl p-2"
        type="number"
        value={form.bend_count}
        onChange={(e) => updateField("bend_count", Number(e.target.value))}
        placeholder="Bend Count"
      />

      <input
        className="border rounded-xl p-2"
        type="number"
        value={form.section_width_mm}
        onChange={(e) => updateField("section_width_mm", Number(e.target.value))}
        placeholder="Section Width"
      />

      <input
        className="border rounded-xl p-2"
        type="number"
        value={form.section_height_mm}
        onChange={(e) => updateField("section_height_mm", Number(e.target.value))}
        placeholder="Section Height"
      />

      <input
        className="border rounded-xl p-2"
        type="number"
        step="0.01"
        value={form.thickness}
        onChange={(e) => updateField("thickness", Number(e.target.value))}
        placeholder="Thickness"
      />

      <select
        className="border rounded-xl p-2"
        value={form.material}
        onChange={(e) => updateField("material", e.target.value)}
      >
        <option value="GI">GI</option>

```javascript
function badgeColor(status) {
  if (status === "pass") return "bg-green-100 text-green-700";
  if (status === "fail") return "bg-red-100 text-red-700";
  if (status === "not_run") return "bg-gray-100 text-gray-700";
  return "bg-yellow-100 text-yellow-700";
}

export default function PipelineStatus({ debugResult }) {
  const stages = debugResult?.stage_debug || [];

  return (
    <div className="rounded-2xl border p-4 shadow-sm bg-white">
      <div className="text-lg font-semibold mb-3">Pipeline Status</div>

      {debugResult?.first_failed_stage && (
        <div className="mb-3 rounded-xl border border-red-200 bg-red-50 p-3 text-red-700">
          First failed stage: {debugResult.first_failed_stage}
        </div>
      )}

      <div className="grid gap-2">
        {stages.map((stage) => (
          <div
            key={stage.stage}
            className="flex items-center justify-between rounded-xl border p-3"
          >
            <div>
              <div className="font-medium">{stage.stage}</div>
              {stage.reason && (
                <div className="text-sm text-gray-600">{stage.reason}</div>
              )}
            </div>
            <span className={`px-3 py-1 rounded-full text-sm ${badgeColor(stage.status)}`}>
              {stage.status}
            </span>
          </div>
        ))}
      </div>
    </div>
  );
}
```

# `pages/Dashboard.jsx`

```javascript
import { useState } from "react";
import UploadForm from "../components/UploadForm";
import PipelineStatus from "../components/PipelineStatus";
import SummaryCards from "../components/SummaryCards";
import EngineDetails from "../components/EngineDetails";
import WarningPanel from "../components/WarningPanel";
import ReportPreview from "../components/ReportPreview";
import ActionButtons from "../components/ActionButtons";
import { runManualModeDebug, runManualTests, exportManualPdf } from "../services/api";

export default function Dashboard() {
  const [loading, setLoading] = useState(false);
  const [payload, setPayload] = useState(null);
  const [pipelineResult, setPipelineResult] = useState(null);
  const [debugResult, setDebugResult] = useState(null);
  const [testResult, setTestResult] = useState(null);
  const [pdfResult, setPdfResult] = useState(null);

  async function handleRun(form) {
    setLoading(true);
    setPayload(form);
    setPdfResult(null);

    try {
      const data = await runManualModeDebug(form);
      setPipelineResult(data.pipeline_result || null);
      setDebugResult(data.debug_result || null);
    } finally {
      setLoading(false);
    }
  }

  async function handleRunTests() {
    setLoading(true);
    try {
      const data = await runManualTests();
      setTestResult(data);
    } finally {
      setLoading(false);
    }
  }

  async function handleExportPdf() {
    if (!payload) return;

    setLoading(true);
    try {
      const data = await exportManualPdf(payload);
      setPdfResult(data);
    } finally {
      setLoading(false);
    }
  }

  const reportEngine = pipelineResult?.report_engine || {};
  const summary = reportEngine?.engineering_summary || {};
  const warnings = reportEngine?.export_json?.warnings || [];
  const assumptions = reportEngine?.export_json?.assumptions || [];
  const readableReport = reportEngine?.readable_report || "";

  return (
    <div className="min-h-screen bg-gray-50 p-4 md:p-6">
      <div className="mx-auto max-w-7xl grid gap-4">
        <div className="text-2xl font-bold">Roll Forming Debug Dashboard</div>

        <UploadForm onRun={handleRun} />

        <ActionButtons
          onRunTests={handleRunTests}
          onExportPdf={handleExportPdf}
          loading={loading}
        />

        <PipelineStatus debugResult={debugResult} />

        <SummaryCards summary={summary} />

        <WarningPanel warnings={warnings} assumptions={assumptions} />

        <ReportPreview reportText={readableReport} />

        <EngineDetails result={pipelineResult} />

        {testResult && (
          <div className="rounded-2xl border p-4 shadow-sm bg-white">
            <div className="text-lg font-semibold mb-3">Test Suite Result</div>
            <pre className="whitespace-pre-wrap text-sm overflow-auto">
              {JSON.stringify(testResult, null, 2)}
            </pre>
          </div>
        )}

        {pdfResult && (
          <div className="rounded-2xl border p-4 shadow-sm bg-white">
            <div className="text-lg font-semibold mb-3">PDF Export Result</div>
            <pre className="whitespace-pre-wrap text-sm overflow-auto">
              {JSON.stringify(pdfResult, null, 2)}
            </pre>
          </div>
        )}
      </div>
    </div>
  );
}
```

# Replit ko exact command

```text
Create a frontend debug dashboard for the engineering app.
Use separate React components for:
1. input panel
2. pipeline status
3. engineering summary
4. engine details
5. warnings and assumptions
6. readable report preview
7. action buttons

Connect the dashboard to:
- POST /api/manual-mode-debug
- GET /api/run-manual-tests
- POST /api/manual-mode-export-pdf

The dashboard must clearly show:
- pass/fail per engine
- first failed stage
- engineering summary cards
- readable report
- raw engine JSON
- test results
- PDF export result

Do not hide backend failures.
Show exact reasons when a stage fails.
```
